\begin{exercise}
	Let $f$ be defined for all real $x$, and suppose that
	\begin{equation*}
	\left| f(x) - f(y) \right| \leq \left( x - y \right)^2
	\end{equation*}
	for all real $x$ and $y$. Prove that $f$ is constant.
\end{exercise}
