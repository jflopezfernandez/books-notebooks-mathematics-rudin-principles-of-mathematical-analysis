\newglossaryentry{open cover}{
	name={open cover},
	description={An \textit{open cover} of a set $E$ in a metric space $X$ is a collection $\left\lbrace G_\alpha \right\rbrace$ of open subsets of $X$ such that $E \subset \cup_\alpha G_\alpha$.}
}

\newglossaryentry{compact}{
	name={compact},
	description={A subset $K$ of a metric space $X$ is said to be \textit{compact} if every open cover of $K$ contains a \textit{finite} subcover.}
}
